# Projeto 8 - IA Generativa Multimodal com Agentic RAG e LangGraph Para Análise Contábil

# Para executar a app:

# Abra o terminal ou prompt de comando, navegue até a pasta com os arquivos e execute o comando abaixo para criar um ambiente virtual:

conda create --name dsaragp8 python=3.12

# Ative o ambiente:

conda activate dsaragp8 (ou: source activate dsaragp8)

# Instale o pip e as dependências:

conda install pip
pip install -r requirements.txt 

# Abra o terminal ou prompt de comando, navegue até a pasta com os arquivos e execute: 

python dsa_p8_setup_rag.py
streamlit run dsa_p8_streamlit_app.py

# Perguntas de exemplo:

# Esta nota fiscal apresenta alguma anomalia?
# ICMS é o único imposto a ser pago com o produto dessa nota?

# Use os comandos abaixo para desativar o ambiente virtual e remover o ambiente (opcional):

conda deactivate
conda remove --name dsaragp8 --all
